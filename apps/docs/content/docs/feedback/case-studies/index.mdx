## Add a 'Report bug' button [#report-bug-button]

> _Stage-A stub._

```tsx
"use client";
import { shipeasy } from "@shipeasy/sdk/client";

export function ReportBug() {
  return (
    <button
      onClick={async () => {
        const title = window.prompt("What's broken?");
        if (!title) return;
        await shipeasy.feedback.bug({ title, pageUrl: window.location.href });
        alert("Thanks!");
      }}
    >
      Report bug
    </button>
  );
}
```
